class LabEquipmentRack:
    """
    Dr. Grace's Lab Equipment Management System
    An array-based storage system for scientific instruments
    """
    
    def __init__(self, capacity=10):
        self.equipment = [None] * capacity  # Pre-allocated array
        self.size = 0
        self.capacity = capacity
        print(f"üî¨ Lab rack initialized with {capacity} positions")
    
    def add_equipment(self, item, position=None):
        """Add equipment to specific position or next available slot"""
        if position is None:
            # Add to end (next available position)
            if self.size < self.capacity:
                self.equipment[self.size] = item
                print(f"‚úÖ Added '{item}' at position {self.size}")
                self.size += 1
            else:
                print("‚ùå Lab rack is full!")
        else:
            # Add to specific position
            if 0 <= position < self.capacity:
                if self.equipment[position] is None:
                    self.equipment[position] = item
                    self.size += 1
                    print(f"‚úÖ Added '{item}' at position {position}")
                else:
                    print(f"‚ö†Ô∏è Position {position} already occupied by '{self.equipment[position]}'")
            else:
                print("‚ùå Invalid position!")
    
    def get_equipment(self, position):
        """Access equipment by position - O(1) operation"""
        if 0 <= position < self.capacity:
            item = self.equipment[position]
            if item:
                print(f"üîç Found '{item}' at position {position}")
                return item
            else:
                print(f"üì≠ Position {position} is empty")
                return None
        else:
            print("‚ùå Position out of range!")
            return None
    
    def search_equipment(self, item):
        """Search for equipment by name - O(n) operation"""
        print(f"üîé Searching for '{item}'...")
        for i in range(self.capacity):
            if self.equipment[i] == item:
                print(f"‚úÖ Found '{item}' at position {i}")
                return i
        print(f"‚ùå '{item}' not found in lab")
        return -1
    
    def remove_equipment(self, position):
        """Remove equipment from specific position"""
        if 0 <= position < self.capacity and self.equipment[position]:
            item = self.equipment[position]
            self.equipment[position] = None
            self.size -= 1
            print(f"üóëÔ∏è Removed '{item}' from position {position}")
            return item
        else:
            print("‚ùå Nothing to remove at that position")
            return None

class EmergencySupplyTower:
    """
    Dr. Grace's Emergency Supply Management System
    A stack-based storage system following LIFO principle
    """
    
    def __init__(self):
        self.supplies = []  # Using Python list as underlying structure
        self.max_height = 10  # Safety limit for our tower
        print("üóº Emergency Supply Tower initialized")
    
    def push(self, supply):
        """Add supply to top of tower - O(1) operation"""
        if len(self.supplies) < self.max_height:
            self.supplies.append(supply)
            print(f"‚¨ÜÔ∏è Pushed '{supply}' to top of tower")
            print(f"üìè Tower height: {len(self.supplies)}")
        else:
            print("‚ö†Ô∏è Tower at maximum height! Cannot add more supplies.")
    
    def pop(self):
        """Remove and return supply from top of tower - O(1) operation"""
        if not self.is_empty():
            supply = self.supplies.pop()
            print(f"‚¨áÔ∏è Popped '{supply}' from top of tower")
            print(f"üìè Tower height: {len(self.supplies)}")
            return supply
        else:
            print("‚ùå Tower is empty! Nothing to pop.")
            return None
    
    def peek(self):
        """Look at top supply without removing it - O(1) operation"""
        if not self.is_empty():
            top_supply = self.supplies[-1]
            print(f"üëÄ Top supply: '{top_supply}'")
            return top_supply
        else:
            print("üì≠ Tower is empty! Nothing to peek at.")
            return None
    
    def is_empty(self):
        """Check if tower is empty"""
        return len(self.supplies) == 0
    
    def size(self):
        """Get current height of tower"""
        return len(self.supplies)
    
    def display_tower(self):
        """Display the entire tower from top to bottom"""
        if self.is_empty():
            print("üì≠ Tower is empty")
        else:
            print("üóº Current Emergency Supply Tower:")
            for i in range(len(self.supplies) - 1, -1, -1):
                level = len(self.supplies) - i
                print(f"Level {level}: [{self.supplies[i]}]")
            print("‚¨áÔ∏è Base of Tower ‚¨áÔ∏è")

from collections import deque

class TaskProcessingQueue:
    """
    Dr. Grace's Task Management System
    A queue-based system following FIFO principle for fair task processing
    """
    
    def __init__(self):
        self.tasks = deque()  # Using deque for efficient operations at both ends
        self.max_capacity = 15  # Prevent system overload
        print("üö∂‚Äç‚ôÇÔ∏è Task Processing Queue initialized")
    
    def enqueue(self, task):
        """Add task to rear of queue - O(1) operation"""
        if len(self.tasks) < self.max_capacity:
            self.tasks.append(task)
            print(f"‚û°Ô∏è Enqueued '{task}' to rear of queue")
            print(f"üìè Queue length: {len(self.tasks)}")
        else:
            print("‚ö†Ô∏è Queue at maximum capacity! Task rejected.")
    
    def dequeue(self):
        """Remove and return task from front of queue - O(1) operation"""
        if not self.is_empty():
            task = self.tasks.popleft()  # Remove from front
            print(f"‚¨ÖÔ∏è Dequeued '{task}' from front of queue")
            print(f"üìè Queue length: {len(self.tasks)}")
            return task
        else:
            print("‚ùå Queue is empty! No tasks to process.")
            return None
    
    def peek(self):
        """Look at front task without removing it - O(1) operation"""
        if not self.is_empty():
            front_task = self.tasks[0]
            print(f"üëÄ Next task to process: '{front_task}'")
            return front_task
        else:
            print("üì≠ Queue is empty! No tasks waiting.")
            return None
    
    def is_empty(self):
        """Check if queue is empty"""
        return len(self.tasks) == 0
    
    def size(self):
        """Get current queue length"""
        return len(self.tasks)
    
    def display_queue(self):
        """Display the entire queue from front to rear"""
        if self.is_empty():
            print("üì≠ No tasks in queue")
        else:
            print("üö∂‚Äç‚ôÇÔ∏è Current Task Processing Queue:")
            print("FRONT ‚û°Ô∏è", end=" ")
            for i, task in enumerate(self.tasks):
                print(f"[{task}]", end="")
                if i < len(self.tasks) - 1:
                    print(" ‚û°Ô∏è ", end="")
            print(" ‚¨ÖÔ∏è REAR")
    
    def process_all_tasks(self):
        """Process all tasks in queue order"""
        print("üîÑ Processing all tasks in queue...")
        while not self.is_empty():
            task = self.dequeue()
            print(f"‚úÖ Processing task: '{task}'")
            # Simulate task processing time
            import time
            time.sleep(0.1)
        print("üéâ All tasks completed!")

class TaskProcessingQueue {
    /**
     * Dr. Grace's Task Management System
     * A queue-based system following FIFO principle for fair task processing
     */
    
    constructor() {
        this.tasks = [];  // Using array with front/rear pointers for efficiency
        this.front = 0;   // Points to front element
        this.rear = 0;    // Points to next insertion position
        this.maxCapacity = 15;  // Prevent system overload
        console.log("üö∂‚Äç‚ôÇÔ∏è Task Processing Queue initialized");
    }
    
    enqueue(task) {
        // Add task to rear of queue - O(1) operation
        if (this.size() < this.maxCapacity) {
            this.tasks[this.rear] = task;
            this.rear++;
            console.log(`‚û°Ô∏è Enqueued '${task}' to rear of queue`);
            console.log(`üìè Queue length: ${this.size()}`);
        } else {
            console.log("‚ö†Ô∏è Queue at maximum capacity! Task rejected.");
        }
    }
    
    dequeue() {
        // Remove and return task from front of queue - O(1) operation
        if (!this.isEmpty()) {
            const task = this.tasks[this.front];
            this.tasks[this.front] = undefined;  // Clear reference
            this.front++;
            console.log(`‚¨ÖÔ∏è Dequeued '${task}' from front of queue`);
            console.log(`üìè Queue length: ${this.size()}`);
            
            // Reset pointers if queue becomes empty
            if (this.isEmpty()) {
                this.front = 0;
                this.rear = 0;
            }
            
            return task;
        } else {
            console.log("‚ùå Queue is empty! No tasks to process.");
            return null;
        }
    }
    
    peek() {
        // Look at front task without removing it - O(1) operation
        if (!this.isEmpty()) {
            const frontTask = this.tasks[this.front];
            console.log(`üëÄ Next task to process: '${frontTask}'`);
            return frontTask;
        } else {
            console.log("üì≠ Queue is empty! No tasks waiting.");
            return null;
        }
    }
    
    isEmpty() {
        // Check if queue is empty
        return this.front >= this.rear;
    }
    
    size() {
        // Get current queue length
        return Math.max(0, this.rear - this.front);
    }
    
    displayQueue() {
        // Display the entire queue from front to rear
        if (this.isEmpty()) {
            console.log("üì≠ No tasks in queue");
        } else {
            console.log("üö∂‚Äç‚ôÇÔ∏è Current Task Processing Queue:");
            let queueString = "FRONT ‚û°Ô∏è ";
            for (let i = this.front; i < this.rear; i++) {
                queueString += `[${this.tasks[i]}]`;
                if (i < this.rear - 1) {
                    queueString += " ‚û°Ô∏è ";
                }
            }
            queueString += " ‚¨ÖÔ∏è REAR";
            console.log(queueString);
        }
    }
    
    async processAllTasks() {
        // Process all tasks in queue order
        console.log("üîÑ Processing all tasks in queue...");
        while (!this.isEmpty()) {
            const task = this.dequeue();
            console.log(`‚úÖ Processing task: '${task}'`);
            // Simulate task processing time
            await new Promise(resolve => setTimeout(resolve, 100));
        }
        console.log("üéâ All tasks completed!");
    }
}
